\section {正規乱数の平均と分散 1}


\begin{verbatim}

\end{verbatim}

\begin{Schunk}
\begin{Sinput}
> set.seed(123); x <- rnorm(100)
> mean(x); var (x)
\end{Sinput}
\begin{Soutput}
[1] 0.09040591
\end{Soutput}
\begin{Soutput}
[1] 0.8332328
\end{Soutput}
\end{Schunk}
すなわち平均0.09
は分散0.83 です。
